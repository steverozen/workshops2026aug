Checkpoint files

Several steps below (CreateChromatinAssay, NucleosomeSignal, TSSEnrichment, GeneActivity, FindTransferAnchors / TransferData) each take many minutes. We save the pbmc Seurat object to .rds checkpoint files at three points so that subsequent runs, including live lecture demos, load the cached object instead of recomputing.

To force recomputation, delete the corresponding .rds file under gitignore/.

ck_dir <- here::here("gitignore")
dir.create(ck_dir, showWarnings = FALSE)

ck_dimred  <- file.path(ck_dir, "pbmc_atac_ck2_dimred.rds")   # post-UMAP + GeneActivity
ck_labeled <- file.path(ck_dir, "pbmc_atac_ck3_labeled.rds")  # post-label-transfer + relabel

Build the Signac object and compute QC metrics

This block builds the Seurat/Signac object from the 10x outputs, keeps standard chromosomes only, attaches Ensembl 98 gene annotations, and computes the nucleosome signal, TSS enrichment, % reads in peaks, and blacklist ratio per cell. The resulting object is cached to ck_qc.

if (file.exists(ck_qc)) {
  pbmc <- readRDS(ck_qc)
  message("Loaded pbmc from ", ck_qc)
} else {
  counts <- Read10X_h5(
    filename = here::here(
      "gitignore/10k_pbmc_ATACv2_nextgem_Chromium_Controller_filtered_peak_bc_matrix.h5"
    )
  )

  metadata <- read.csv(
    file = here::here("10k_pbmc_ATACv2_nextgem_Chromium_Controller_singlecell.csv"),
    header = TRUE,
    row.names = 1
  )

  chrom_assay <- CreateChromatinAssay(
    counts = counts,
    sep = c(":", "-"),
    fragments = here::here(
      "gitignore/10k_pbmc_ATACv2_nextgem_Chromium_Controller_fragments.tsv.gz"
    ),
    min.cells = 10,
    min.features = 200
  )

  pbmc <- CreateSeuratObject(
    counts = chrom_assay,
    assay = "peaks",
    meta.data = metadata
  )

  # Keep only standard chromosomes.
  peaks.keep <- seqnames(granges(pbmc)) %in% standardChromosomes(granges(pbmc))
  pbmc <- pbmc[as.vector(peaks.keep), ]

  # Pull Ensembl 98 EnsDb (Homo sapiens) and the ENCODE hg38 blacklist
  # from AnnotationHub. `AH75011` is the accession for the Ensembl 98
  # EnsDb (a SQLite-backed Ensembl annotation database mapped to GRCh38).
  ah <- AnnotationHub()
  ensdb_v98 <- ah[["AH75011"]]
  blacklist_regions <- ah[["AH107305"]] # ENCODE blacklist for hg38

  annotations <- GetGRangesFromEnsDb(ensdb = ensdb_v98)
  seqlevels(annotations) <- paste0("chr", seqlevels(annotations))
  genome(annotations) <- "hg38"
  Annotation(pbmc) <- annotations

  pbmc <- NucleosomeSignal(object = pbmc)
  pbmc <- TSSEnrichment(object = pbmc)
  pbmc$pct_reads_in_peaks <- pbmc$peak_region_fragments / pbmc$passed_filters * 100
  pbmc$blacklist_ratio <- FractionCountsInRegion(
    object = pbmc, assay = "peaks", regions = blacklist_regions
  )

  saveRDS(pbmc, ck_qc)
  message("Saved checkpoint ", ck_qc)
}

if (file.exists(ck_dimred)) {
  pbmc <- readRDS(ck_dimred)
  message("Loaded pbmc from ", ck_dimred)
} else {
  pbmc <- subset(
    x = pbmc,
    subset = nCount_peaks > 9000 &
      nCount_peaks < 100000 &
      pct_reads_in_peaks > 40 &
      blacklist_ratio < 0.01 &
      nucleosome_signal < 4 &
      TSS.enrichment > 4
  )

  pbmc <- RunTFIDF(pbmc)
  pbmc <- FindTopFeatures(pbmc, min.cutoff = "q0")
  pbmc <- RunSVD(pbmc)

  pbmc <- RunUMAP(object = pbmc, reduction = "lsi", dims = 2:30)
  pbmc <- FindNeighbors(object = pbmc, reduction = "lsi", dims = 2:30)
  pbmc <- FindClusters(object = pbmc, verbose = FALSE, algorithm = 3)

  gene.activities <- GeneActivity(pbmc)
  pbmc[["RNA"]] <- CreateAssayObject(counts = gene.activities)
  pbmc <- NormalizeData(
    object = pbmc, assay = "RNA",
    normalization.method = "LogNormalize",
    scale.factor = median(pbmc$nCount_RNA)
  )

  saveRDS(pbmc, ck_dimred)
  message("Saved checkpoint ", ck_dimred)
}
