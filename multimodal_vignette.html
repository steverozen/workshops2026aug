# Load in the RNA UMI matrix.
#
# Note that this dataset also contains ~5% of mouse cells, which we can use as
# negative controls for the protein measurements. For this reason, the gene
# expression matrix has HUMAN_ or MOUSE_ appended to the beginning of each
# gene.
cbmc.rna <- as.sparse(read.csv(
  file = here::here("GSE100866_CBMC_8K_13AB_10X-RNA_umi.csv.gz"),
  sep = ",", header = TRUE, row.names = 1
))

# To make life a bit easier going forward, we discard all but the top 100 most
# highly expressed mouse genes, and remove the HUMAN_ prefix from the human
# genes.
cbmc.rna <- CollapseSpeciesExpressionMatrix(cbmc.rna) # Seurat function

# Load in the ADT UMI matrix (ADT = antibody derived tags).
cbmc.adt <- as.sparse(read.csv(
  file = here::here("GSE100866_CBMC_8K_13AB_10X-ADT_umi.csv.gz"),
  sep = ",", header = TRUE, row.names = 1
))

# Since measurements were made in the same cells, the two matrices have
# identical column names.
all.equal(colnames(cbmc.rna), colnames(cbmc.adt))
